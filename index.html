Connect your Sheet

You'll need two things: your Google Sheet ID (used as your private key) and a Google Apps Script URL (handles all read/write).

From your Sheet URL: docs.google.com/spreadsheets/d/THIS_PART/edit
See setup guide for how to create this (free, ~5 min). Show guide ↓
Don't have a sheet? Create one here, shared as "Anyone with link can edit".

Apps Script Setup (5 min)

1. Open your Google Sheet → Extensions → Apps Script

2. Delete any existing code, paste the script below

3. Click Deploy → New deployment → Web App

4. Set "Execute as" = Me, "Who has access" = Anyone

5. Copy the Web App URL and paste it above

Click to copy the Apps Script code ↓

function doGet(e){return handle(e)}
function doPost(e){return handle(e)}
function handle(e){
  try{
    var p=e.parameter,ss=SpreadsheetApp.getActiveSpreadsheet();
    var sh=ss.getSheetByName(p.sheet)||ss.insertSheet(p.sheet);
    if(p.action==='get'){
      return out(sh.getDataRange().getValues());
    }else if(p.action==='append'){
      sh.appendRow(JSON.parse(p.row));return out('ok');
    }else if(p.action==='rewrite'){
      sh.clearContents();
      var rows=JSON.parse(p.rows);
      if(rows.length)sh.getRange(1,1,rows.length,rows[0].length).setValues(rows);
      return out('ok');
    }}
  }catch(err){return out({error:err.message})}
}
function out(d){
  return ContentService.createTextOutput(JSON.stringify(d))
    .setMimeType(ContentService.MimeType.JSON);
}
